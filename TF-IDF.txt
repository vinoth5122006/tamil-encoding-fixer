#TRAINING PART

import pandas as pd
import pickle
from sklearn.feature_extraction.text import TfidfVectorizer
from google.colab import files
import re
print("Upload training dataset (Excel or CSV)...")
uploaded = files.upload()
file = list(uploaded.keys())[0]
df = pd.read_excel(file) if file.endswith(".xlsx") else pd.read_csv(file)

# Normalize text

def normalize(text):
    if pd.isna(text): return ""
    text = str(text).lower().strip()
    text = re.sub(r'\s+', ' ', text)
    return text
df["Normalized_Tanglish"] = df["Tanglish Sentence"].apply(normalize)
df["Normalized_Context"]  = df["Previous Chat(Context)"].apply(normalize)
df["Combined"] = df["Normalized_Context"] + " " + df["Normalized_Tanglish"]

# Build TF-IDF vectorizer

vectorizer = TfidfVectorizer(max_features=500, ngram_range=(1,3))
vectorizer.fit(df["Combined"])

# Build phonetic map (from dataset)

PHONETIC = {}
for _, row in df.iterrows():
    word = str(row["Ambiguous Word"]).lower().strip()
    tamil = str(row["Correct Tamil Output"]).strip()
    if word and tamil:
        PHONETIC[word] = tamil

# SAVE ALL FILES

with open("phonetic_map.pkl","wb") as f:
    pickle.dump(PHONETIC,f)

with open("tfidf.pkl","wb") as f:
    pickle.dump(vectorizer,f)

df.to_csv("train_processed.csv",index=False)

print("\n TRAINING COMPLETE!")
print("Generated files:")
print(" - phonetic_map.pkl")
print(" - tfidf.pkl")
print(" - train_processed.csv")
files.download("phonetic_map.pkl")
files.download("tfidf.pkl")
files.download("train_processed.csv")
